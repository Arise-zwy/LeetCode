## 617.合并二叉树
```cpp
class Solution {
public:
    void process(TreeNode* &t1, TreeNode* &t2){
        if(t1 == nullptr && t2 == nullptr){
            return;
        }
        if(t1 == nullptr){
            t1 = new TreeNode(0);   
        }
        if(t2 == nullptr){
            t2 = new TreeNode(0);
        }
        t1->val += t2->val;

        process(t1->left, t2->left);
        process(t1->right, t2->right);
    }
    TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        process(t1, t2);
        return t1;
    }    
};
```
## 102.二叉树层序遍历
```cpp
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if(root == nullptr)
        {
		    return res;
	    }
        queue<TreeNode*> Q;
        TreeNode* p;
        Q.push(root);
        while(Q.empty()==0)
        {
            vector<int> a;
            int width=Q.size(); // 这句是分层存储的重点
            for(int i=0;i<width;i++)
            {
                p=Q.front();
                a.push_back(p->val);
                Q.pop();
                if(p->left) Q.push(p->left);
                if(p->right) Q.push(p->right);
            }
            res.push_back(a);
        }
        return res;
    }
};
```
