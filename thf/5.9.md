## 面试题42.连续子数组的最大和
```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int maxsub = nums[0];
        int sum = INT_MIN;
        for(int i = 0; i < nums.size(); ++i){
            if(sum < 0){
                sum = nums[i];
            }
            else{
                sum += nums[i]; 
            }
            maxsub = max(sum, maxsub);
        }
        return maxsub;
    }
};
```

## 面试题38.字符串的排列
```cpp
class Solution {
public:
    vector<string> ans;
    vector<bool> visit;
    string track;
    vector<string> permutation(string s) {
        if(s.empty()) {
            return{};
        }
        for(int i = 0; i < s.size(); ++i){
            visit.push_back(false);
        }
        backtrack(s);
        sort(ans.begin(),ans.end());
        ans.erase(unique(ans.begin(),ans.end()),ans.end());
        return ans;
    }
    void backtrack(const string &s){
        if(track.size() == s.size()){
            ans.push_back(track);
            return;
        }
        for(int i = 0; i < s.size(); ++i){
            if(visit[i]){
                continue;
            }
            visit[i] = true;
            track.push_back(s[i]);
            backtrack(s);
            visit[i] = false;
            track.pop_back();
        }
    }
};
```
